import numpy as np

def main(data: dict) -> np.ndarray:
    prices = data["price"]            # [N, 20]
    current = prices[:, -1]           # [N]
    mean = prices.mean(axis=1)        # [N]
    safe_mean = np.where(mean != 0, mean, 1.0)
    deviation = (current - safe_mean) / safe_mean
    return deviation  # auto-normalized
